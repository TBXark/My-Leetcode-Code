# Solution

class Solution {
    func myPow(_ x: Double, _ n: Int) -> Double {
                if n == 0 {
            return 1
        } else if n > 2 {
            let p = myPow(x, n / 2)
            let q = myPow(x, n % 2)
            return q * p * p
        } else if n > 0 {
            return x * myPow(x, n - 1)
        } else {
            return myPow(1/x, -n)
        }

    }
}
